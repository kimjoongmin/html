``` javascript
  class Car{
    constructor(color){
      this.wheels = 4;
      this.color = color;
    }
    drive(){
      console.log("drive..");
    }
  }
  class Bmw extends Car {
    constructor(color){
      super(color);
      this.navigation = 1;
    }
    drive(){
      super.drive();
      console.log("slow~");
    }
    stop(){
      console.log("stop!!");
    }
  }
  const x5 = new Bmw("black");
```
