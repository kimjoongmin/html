    *전환 효과 속성 적용
    순서에 따라~
    .box{
      transition-property:background-color;
      transition-duration:1s;
      transition-timing-function:linear;
      transition-delay:3s;
      
      transition:all 1s ease 3s;
    }
    


    *애니메이션 속성으로 전환~
    transition 보다 더 부드러움
    .box{
      animation-name : bgChange;
      animation-duration : 2s;
      animation-delay : 2s;
      /*변한 후 유지*/
      animation-fill-mode: none; /* 자기자신 고유 css 가지고 있음 */
      animation-fill-mode: forwards; /* 종료 지점의 css 유지함 */
      animation-fill-mode: backwards; /* 애니메이션 0%의 css 가지고 있음 */
      animation-fill-mode: both; /* 키프레임안의 css를 적용 및 유지하고 있음 */
      animation-iteration-count: infinite;
      animation-play-state: running;/*기본값*/
      animation-direction: alternate; /*홀수번째는 순방향 짝수번째는 역방향으로 애니메이션*/
      animation-timing-function: ease;
      animation:bgChange 2s 2s forwards infinite;
    }
    .box:hover{
      animation-play-state: paused;/*오버했을때 애니메이션 멈춤*/
    }
    @keyframes bgChange {
      0%{ 
        background-color:blue;
      }
      100%{
        background-color:orange;
      }
    }
